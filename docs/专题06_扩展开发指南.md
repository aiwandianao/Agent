#### 步骤 3：实现工具

```java
package com.dasi.mcp.feishu.tool;

import com.alibaba.fastjson2.JSON;
import lombok.Data;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Component;
import org.springframework.web.client.RestTemplate;

/**
 * 飞书通知工具
 */
@Slf4j
@Component
public class FeishuTool {

    private final RestTemplate restTemplate = new RestTemplate();

    /**
     * 发送文本消息
     */
    @Tool(description = "发送飞书文本消息")
    public FeishuResponse sendText(
            @P("Webhook URL") String webhookUrl,
            @P("消息内容") String content
    ) {
        try {
            // 构建请求体
            FeishuTextMessage message = new FeishuTextMessage();
            message.getMsg_type().set("text");
            message.getContent().set("text", content);

            // 发送请求
            String response = restTemplate.postForObject(
                    webhookUrl,
                    message,
                    String.class
            );

            log.info("飞书消息发送成功: {}", response);

            return FeishuResponse.builder()
                    .success(true)
                    .message("消息已发送")
                    .build();

        } catch (Exception e) {
            log.error("飞书消息发送失败", e);
            return FeishuResponse.builder()
                    .success(false)
                    .message("发送失败: " + e.getMessage())
                    .build();
        }
    }

    /**
     * 发送卡片消息
     */
    @Tool(description = "发送飞书卡片消息")
    public FeishuResponse sendCard(
            @P("Webhook URL") String webhookUrl,
            @P("标题") String title,
            @P("内容") String content
    ) {
        try {
            FeishuCardMessage message = new FeishuCardMessage();
            message.getMsg_type().set("interactive");
            message.getCard().getHeader().setTitle(title);
            message.getCard().getElements().add(
                    new FeishuCardMessage.CardElement("div", content)
            );

            String response = restTemplate.postForObject(
                    webhookUrl,
                    message,
                    String.class
            );

            return FeishuResponse.builder()
                    .success(true)
                    .message("卡片已发送")
                    .build();

        } catch (Exception e) {
            return FeishuResponse.builder()
                    .success(false)
                    .message("发送失败: " + e.getMessage())
                    .build();
        }
    }

    @Data
    static class FeishuTextMessage {
        private Map<String, String> msg_type = new HashMap<>();
        private Map<String, String> content = new HashMap<>();
    }

    @Data
    static class FeishuCardMessage {
        private Map<String, String> msg_type = new HashMap<>();
        private Card card = new Card();

        @Data
        static class Card {
            private Header header = new Header();
            private List<CardElement> elements = new ArrayList<>();

            @Data
            static class Header {
                private String title;
            }

            @Data
            static class CardElement {
                private String tag;
                private String text;

                public CardElement(String tag, String text) {
                    this.tag = tag;
                    this.text = text;
                }
            }
        }
    }

    @Data
    @Builder
    static class FeishuResponse {
        private boolean success;
        private String message;
    }
}
```

## 四、新增工作流策略

### 4.1 业务场景

**场景**：需要一个"并行执行"策略，多个节点同时执行

### 4.2 操作步骤

#### 步骤 1：定义策略接口

```java
public interface ExecuteStrategy {

    /**
     * 执行工作流
     */
    void execute(ExecuteRequestEntity request, ExecuteContext context);

    /**
     * 获取策略类型
     */
    String getType();
}
```

#### 步骤 2：实现策略

```java
@Component("parallelStrategy")
public class ExecuteParallelStrategy implements ExecuteStrategy {

    @Override
    public void execute(ExecuteRequestEntity request, ExecuteContext context) {
        // 1. 获取并行执行的节点列表
        List<AbstractExecuteNode> nodes = getParallelNodes(request.getAgentId());

        // 2. 创建并行任务
        List<CompletableFuture<Void>> futures = nodes.stream()
                .map(node -> CompletableFuture.runAsync(() -> {
                    try {
                        node.apply(request, context);
                    } catch (Exception e) {
                        log.error("节点执行失败: {}", node.getClass().getSimpleName(), e);
                    }
                }, executorService))
                .collect(Collectors.toList());

        // 3. 等待所有节点完成
        CompletableFuture.allOf(futures.toArray(new CompletableFuture[0])).join();

        // 4. 汇总结果
        summarizeResults(context);
    }

    @Override
    public String getType() {
        return "parallel";
    }
}
```

#### 步骤 3：注册到工厂

```java
@Component
public class ExecuteStrategyFactory {

    @Resource
    private Map<String, ExecuteStrategy> strategyMap;

    public ExecuteStrategy getStrategy(String agentId) {
        // 从数据库获取 Agent 配置
        AiAgent agent = agentDao.getByAgentId(agentId);

        // 根据策略类型获取对应的实现
        return strategyMap.get(agent.getAgentType() + "Strategy");
    }
}
```

#### 步骤 2：实现 Elasticsearch 向量存储

```java
@Component
@ConditionalOnProperty(name = "vector.store.type", havingValue = "elasticsearch")
public class ElasticsearchVectorStore implements VectorStore {

    @Resource
    private ElasticsearchTemplate template;

    @Override
    public void add(List<Document> documents) {
        List<VectorDocument> vectorDocs = documents.stream()
                .map(this::toVectorDocument)
                .collect(Collectors.toList());

        template.save(vectorDocs);
    }

    @Override
    public List<Document> similaritySearch(SearchRequest request) {
        // 构建向量查询
        NativeSearchQuery query = NativeSearchQuery.builder()
                .withQuery(QueryBuilders.vectorQuery(request.getQuery(), "embedding"))
                .withMaxResults(request.getTopK())
                .build();

        SearchHits<VectorDocument> hits = template.search(query, VectorDocument.class);

        return hits.stream()
                .map(hit -> toDocument(hit.getContent()))
                .collect(Collectors.toList());
    }

    private VectorDocument toVectorDocument(Document doc) {
        VectorDocument vectorDoc = new VectorDocument();
        vectorDoc.setId(UUID.randomUUID().toString());
        vectorDoc.setContent(doc.getText());
        vectorDoc.setEmbedding(doc.getEmbedding());
        vectorDoc.setMetadata(doc.getMetadata());
        return vectorDoc;
    }
}
```

@Component
public class StockTool {

    @Resource
    private RestTemplate restTemplate;

    @Value("{stock.api.key}")
    private String apiKey;

    @Tool(description = "查询股票实时行情")
    public StockResponse queryStock(
            @P("股票代码，如：000001、600000") String stockCode
    ) {
        try {
            // 调用股票 API
            String url = String.format(
                    "https://api.stock.com/quote?key=%s&code=%s",
                    apiKey, stockCode
            );

            String response = restTemplate.getForObject(url, String.class);
            JSONObject json = JSON.parseObject(response);

            return StockResponse.builder()
                    .stockCode(stockCode)
                    .stockName(json.getString("name"))
                    .currentPrice(json.getBigDecimal("currentPrice"))
                    .change(json.getBigDecimal("change"))
                    .changePercent(json.getBigDecimal("changePercent"))
                    .build();

        } catch (Exception e) {
            return StockResponse.builder()
                    .error("查询失败: " + e.getMessage())
                    .build();
        }
    }

    @Data
    @Builder
    static class StockResponse {
        private String stockCode;
        private String stockName;
        private BigDecimal currentPrice;
        private BigDecimal change;
        private BigDecimal changePercent;
        private String error;
    }
}

**Q7**: 设计一个"分支执行"的工作流策略。

<details>
<summary>点击查看答案</summary>

**答案**：
```java
@Component("branchStrategy")
public class ExecuteBranchStrategy implements ExecuteStrategy {

    @Override
    public void execute(ExecuteRequestEntity request, ExecuteContext context) {
        // 1. 执行决策节点
        AbstractExecuteNode decisionNode = getDecisionNode(request.getAgentId());
        String decision = decisionNode.apply(request, context);

        // 2. 解析决策结果
        JSONObject decisionJson = JSON.parseObject(decision);
        String branch = decisionJson.getString("branch");

        // 3. 根据分支执行不同节点
        AbstractExecuteNode branchNode = getBranchNode(request.getAgentId(), branch);
        branchNode.apply(request, context);
    }

    @Override
    public String getType() {
        return "branch";
    }
}
```

**配置示例**：
```
决策节点：
{
  "node": "decision",
  "prompt": "判断用户问题类型（技术/商务/其他）"
}

分支配置：
{
  "technical": "tech_support_node",
  "business": "sales_node",
  "other": "general_node"
}
```
